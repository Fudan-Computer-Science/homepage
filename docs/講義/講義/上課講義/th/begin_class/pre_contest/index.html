#include<iostream>
using namespace std;
int main(){
	int arr[10]={8,4,3,7,9,1,2,6,5,0};
	for(int i=5;i>0;i--){
		for(int j=0;j<i;j++){
			if(arr[j]>arr[j+1]){
				int temp=arr[j];
				arr[j]=arr[j+1];
				arr[j+1]=temp;
			} 
		}
	}
	for(int i=9;i>5;i--){
		for(int j=5;j<i;j++){
			if(arr[j]>arr[j+1]){
				int temp=arr[j];
				arr[j]=arr[j+1];
				arr[j+1]=temp;
			} 
		}
	}
	for(int i=0;i<10;i++){
		cout<<arr[i]<<" "; 
	}	 
	return 0;
}

答案:B​
